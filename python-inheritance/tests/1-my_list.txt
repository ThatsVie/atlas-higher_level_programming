Import function from the module:
	>>> MyList = __import__("1-my_list").MyList

Check for empty list:
	>>> l = MyList()
	>>> l.print_sorted()
	[]

Check for list in correct order:
	>>> l.print_sorted()
	[1, 2, 3, 4]

Check appendng works:
	>>> l.append(1)
	>>> l.append(2)
	>>> l.append(3)
	>>> l.append(4)
	>>> print(l)
	[1, 2, 3, 4]

Check reversed order list:
	>>> l = MyList()
	>>> l.append(4)
	>>> l.append(3)
	>>> l.append(2)
	>>> l.append(1)
	>>> print(l)
	[4, 3, 2, 1]
	>>> l.print_sorted()
	[1, 2, 3, 4]
	>>> print(l)
	[4, 3, 2, 1]
